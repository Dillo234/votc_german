log_new_relations_test = {
	save_scope_as = mcc_tmp_char
	every_in_global_list = {
    	variable = mcc_characters_list_v2
    	if = {
			limit = {
				is_at_same_location = scope:actor
				NOT = { this = prev }
			}
			if = {
				limit = {
					has_relation_to = scope:mcc_tmp_char
				}
				debug_log = "VOTC:IN/;/new_relations/;/[PREV.Char.GetID]/;/[THIS.Char.GetID]/;/yes/;/STARTMULTILINE#[THIS.Char.GetRelationToStringExtended(PREV.Char.Self)]#ENDMULTILINE/;/"
			}
			else = {
				debug_log = "VOTC:IN/;/new_relations/;/[PREV.Char.GetID]/;/[THIS.Char.GetID]/;/no/;/STARTMULTILINE#[THIS.Char.GetRelationToString(PREV.Char.Self)]#ENDMULTILINE/;/"
			}
		}
	}
	scope:actor = {
		if = {
			limit = {
				NOT = {
					is_target_in_global_variable_list = {
						name = mcc_characters_list_v2
						target = this
					}
					this = prev
				}
			}
			if = {
				limit = {
					has_relation_to = scope:mcc_tmp_char
				}
				debug_log = "VOTC:IN/;/new_relations/;/[PREV.Char.GetID]/;/[THIS.Char.GetID]/;/yes/;/STARTMULTILINE#[THIS.Char.GetRelationToStringExtended(PREV.Char.Self)]#ENDMULTILINE/;/"
			}
			else = {
				debug_log = "VOTC:IN/;/new_relations/;/[PREV.Char.GetID]/;/[THIS.Char.GetID]/;/no/;/STARTMULTILINE#[THIS.Char.GetRelationToString(PREV.Char.Self)]#ENDMULTILINE/;/"
			}
		}
	}
	## debug_log = "VOTC:IN/;/new_relations/;/[THIS.Char.GetID]/;/STARTMULTILINE#[THIS.Char.GetRelationToStringExtended(GetPlayer)]#ENDMULTILINE/;/"
}

log_secrets = {
	set_global_variable = {
		name = revotc_mcc_tmp_var
		value = this
	}
	save_scope_as = revotc_mcc_tmp_var
    every_secret = {
		debug_log = "VOTC:IN/;/secret/;/[PREV.Char.GetID]/;/[THIS.Secret.GetNameNoTooltip]/;/[THIS.Secret.GetDesc]/;/[THIS.Secret.GetCategoryName]/;/[GetSecretType(THIS.Secret)]"
		
		# Log this secret is is_shunned_for or is_criminal_for liege, court_owner or province owner
		if = {
			limit = {
				is_criminal_for = prev
			}

			debug_log = "VOTC:IN/;/secret_is_criminal/;/[PREV.Char.GetID]/;/"
		}
		else_if = {
			limit = {
				is_shunned_for = prev
			}
			debug_log = "VOTC:IN/;/secret_is_shunned/;/[PREV.Char.GetID]/;/"
		}

		# Log secret target if exists (personal secrets don't have targets)
		if = {
			limit = {
				exists = secret_target
			}
			secret_target = {
				debug_log = "VOTC:IN/;/secret_target/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/[PREV.Secret.GetNameNoTooltip]/;/[THIS.Char.GetID]/;/[THIS.Char.GetTitledFirstNameNicknamedNoTooltipRegnal]"
			}
		}
		
		# Log all knowers of this secret
		every_secret_knower = {
			save_scope_as = mcc_sec_k
			limit = {
				NOT = { this = scope:revotc_mcc_tmp_var }
			}
			debug_log = "VOTC:IN/;/secret_knower/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/[PREV.Secret.GetNameNoTooltip]/;/[THIS.Char.GetID]/;/[THIS.Char.GetTitledFirstNameNicknamedNoTooltipRegnal]"
			
			# Log if knower spent this secret
			if = {
				limit = {
					prev = {
						is_spent_by = scope:mcc_sec_k
					}
				}
				debug_log = "VOTC:IN/;/secret_spent/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/yes"
			}
			else = {
				debug_log = "VOTC:IN/;/secret_spent/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/no"
			}

			# Log if knower can expose this secret
			if = {
				limit = {
					prev = {
						can_be_exposed_by = scope:mcc_sec_k
					}
				}
				debug_log = "VOTC:IN/;/secret_can_be_exposed/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/yes"
			}
			else = {
				debug_log = "VOTC:IN/;/secret_can_be_exposed/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/no"
			}
		}


		debug_log = "VOTC:IN/;/secret_eob/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/"
    }
}

log_ksecrets = {
		set_global_variable = {
			name = revotc_mcc_tmp_var
			value = this
		}
		save_scope_as = revotc_mcc_tmp_var

    every_known_secret = {
		debug_log = "VOTC:IN/;/k_secret/;/[PREV.Char.GetID]/;/[THIS.Secret.GetNameNoTooltip]/;/[THIS.Secret.GetDesc]/;/[THIS.Secret.GetCategoryName]/;/[GetSecretType(THIS.Secret)]"
		secret_owner = {
			debug_log = "VOTC:IN/;/k_secret_owner/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/[THIS.Char.GetID]/;/[THIS.Char.GetTitledFirstNameNicknamedNoTooltipRegnal]"
		}
		# Log this secret is is_shunned_for or is_criminal_for 
		if = {
			limit = {
				is_criminal_for = secret_owner
			}

			debug_log = "VOTC:IN/;/k_secret_is_criminal/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/"
		}
		else_if = {
			limit = {
				is_shunned_for = secret_owner
			}
			debug_log = "VOTC:IN/;/k_secret_is_shunned/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/"
		}

		# Log secret target if exists (personal secrets don't have targets)
		if = {
			limit = {
				exists = secret_target
			}
			secret_target = {
				debug_log = "VOTC:IN/;/k_secret_target/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/[THIS.Char.GetID]/;/[THIS.Char.GetTitledFirstNameNicknamedNoTooltipRegnal]"
			}
		}
		
		# Log if this character spent this secret
		if = {
			limit = {
					is_spent_by = prev
			}
			debug_log = "VOTC:IN/;/k_secret_spent/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/yes"
		}
		else = {
			debug_log = "VOTC:IN/;/k_secret_spent/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/no"
		}

		# Log if knower can expose this secret
		if = {
			limit = {
					can_be_exposed_by = prev
			}
			debug_log = "VOTC:IN/;/k_secret_can_be_exposed/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/yes"
		}
		else = {
			debug_log = "VOTC:IN/;/k_secret_can_be_exposed/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/no"
		}
		
		# Log all knowers of this secret (for rp might be useful, so character could tell, who told this secret)
		every_secret_knower = {
			limit = {
				NOT = { this = scope:revotc_mcc_tmp_var }
			}
			debug_log = "VOTC:IN/;/k_secret_knower/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/[PREV.Secret.GetNameNoTooltip]/;/[THIS.Char.GetID]/;/[THIS.Char.GetTitledFirstNameNicknamedNoTooltipRegnal]"
		}


		debug_log = "VOTC:IN/;/k_secret_eob/;/[GetGlobalVariable('revotc_mcc_tmp_var').GetCharacter.GetID]/;/"
    }
}
